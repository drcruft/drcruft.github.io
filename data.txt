```python
import os
import time
import splunklib.client as client
from crewai.tools import BaseTool

class SplunkSearchTool(BaseTool):
    name: str = "splunk_search"
    description: str = (
        "Execute a Splunk SPL query and return the JSON results."
    )

    def __init__(self, host: str, port: int, username: str, password: str, **kwargs):
        super().__init__(**kwargs)
        # Establish connection to Splunk REST API
        self.service = client.connect(
            host=host, port=port, username=username, password=password
        )

    def _run(self, query: str, earliest: str = "-15m", latest: str = "now", **kwargs):
        # Create and poll a search job
        job = self.service.jobs.create(query, earliest_time=earliest, latest_time=latest)
        while not job.is_done():
            time.sleep(0.5)
        # Fetch results in JSON
        results = job.results(output_mode="json")
        return results.read()
```
