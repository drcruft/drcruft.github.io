```python
from crewai import Agent, Task, Crew, Process

# Load Splunk credentials from env vars (or hard-code for testing)
HOST     = os.getenv("SPLUNK_HOST", "splunk.example.com")
PORT     = int(os.getenv("SPLUNK_PORT", 8089))
USER     = os.getenv("SPLUNK_USERNAME", "admin")
PASS     = os.getenv("SPLUNK_PASSWORD", "changeme")

# Instantiate the tool
splunk_tool = SplunkSearchTool(
    host=HOST, port=PORT, username=USER, password=PASS
)

# Define an Agent that can use the Splunk tool
splunk_agent = Agent(
    name="SplunkAnalyst",
    tools=[splunk_tool],
    description="Runs SPL queries against Splunk and returns results."
)

# Define a Task that delegates to that Agent
splunk_task = Task(
    name="RunSplunkQuery",
    agent=splunk_agent,
    description="Given an SPL query, execute it via the SplunkSearchTool.",
    tools=[splunk_tool]
)

# Build and run the Crew
crew = Crew(
    agents=[splunk_agent],
    tasks=[splunk_task],
    process=Process.sequential
)

# Kick off the crew with your query
output = crew.kickoff(input="search index=_internal | head 5")
print(output)
```
